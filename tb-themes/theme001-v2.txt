    {block:IfInfiniteScroll}

    function basicInfiniteScroll() {
        
        var currentLoc = location;
        var nextPageNumber = 2;
                
        if(location.pathname.includes('/page/')) {
            let pageNumberPos = location.pathname.search("/page/") + 6;
            let pageNumber = location.pathname.slice(pageNumberPos, location.pathname.length);
            currentLoc = location.origin + location.pathname.slice(0, pageNumberPos - 6);
            nextPageNumber = Number(pageNumber) + 1;
        }; // if
        
        function loadNextPage() {
            const xhttp = new XMLHttpRequest();
            xhttp.onload = function() {
                    
                let nextPageContent = this.responseText;
                let postsStart = nextPageContent.search('<div class="post caixa hasdate');
                    
                var pageContentExists = postsStart < 120000;
                    
                if(pageContentExists) {
                    let postsEnd = nextPageContent.search('</main> <!-- #posts -->');
                    let postsCode = nextPageContent.slice(postsStart, postsEnd);
                    document.querySelector("#posts").insertAdjacentHTML("beforeend", postsCode);
                } else {
                    document.querySelector("#infscr-button").style.display="none";
                    document.querySelector("#infscr-postsend").style.visibility="visible";
                    document.querySelector("#infscr-postsend").style.opacity="1";
                }; // else
                    
            }; // onload
                
            xhttp.open("GET", currentLoc + "/page/" + nextPageNumber);
            xhttp.send();
            nextPageNumber += 1;
                
            // after load functions:
            removeEmptyTextBodies();
            shortenNotes();
            imgsRes();
            wrapNPFPhotosets();
            hideNPFcaptions();
            shortenLongPosts();
            captionsFix();
            masonryGrid();
            translatePage();
            vIcons();
            altPalette();
            
            // after load msg:
            document.querySelector("#infscr-loading").style.visibility="hidden";
            document.querySelector("#infscr-loading").style.opacity="0";
                
        }; // function loadNextPage();
        
        
        {block:IfNotAutomaticInfiniteScroll}
        document.querySelector("#infscr-button").onclick = function() {
            loadNextPage();
        };
        {/block:IfNotAutomaticInfiniteScroll}
        
        
        {block:IfAutomaticInfiniteScroll}
        document.onscroll = function() {
            let pageHeight = document.querySelector("#all").offsetHeight;
            let endOfPage = pageHeight - window.innerHeight - 20;
            // load next page:
            if(window.pageYOffset >= endOfPage) {loadNextPage();};
        }; // onscroll
        {/block:IfAutomaticInfiniteScroll}
        
        
    }; basicInfiniteScroll();
